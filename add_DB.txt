CREATE TABLE Kategooriatable(
	Id int PRIMARY KEY IDENTITY (1,1) NOT NULL,
	Kategooria_nimetus nvarchar(50) NOT NULL,
	Kirjeldus text NULL
)

CREATE TABLE Toodetable(
	Id int PRIMARY KEY IDENTITY(1,1) NOT NULL,
	Toodenimetus varchar(50) NOT NULL,
	Kogus int NOT NULL,
	Hind float NOT NULL,
	Pilt text NOT NULL,
	Kategooria int NOT NULL
	CONSTRAINT [FK_Toodetable_Kategooriatable] FOREIGN KEY ([Kategooria]) REFERENCES [dbo].[Kategooriatable] ([Id])
)

CREATE TABLE Klient(
	Id int PRIMARY KEY IDENTITY (1,1) NOT NULL,
	Nimi varchar(50) NOT NULL,
	Email varchar(50) NOT NULL,
	Password text NOT NULL
)
